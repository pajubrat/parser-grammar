
# Universal lexical redundancy rules

# Basic lexical categories
FORCE	:: -ARG Fin 			!COMP:* !PROBE:Fin COMP:T/fin COMP:C/fin -SPEC:T/fin
C/fin 	:: -ARG Fin C 			-COMP:C/fin -COMP:T/prt COMP:T/fin !PROBE:Fin COMP:Neg/fin !COMP:* -SPEC:N -SPEC:T/fin -SPEC:C/fin -COMP:FORCE
C		:: EF					-SPEC:MA/A 
T/fin 	:: ARG EF +ΦLF,ΦPF T Fin !COMP:* -COMP:a -SPEC:C SPEC:T/prt COMP:v COMP:V COMP:Pass -COMP:Inf -COMP:T/fin !PROBE:V SPEC:φ SPEC:D SPEC:C/fin -SPEC:T/fin -SPEC:Neg SPEC:FORCE -SPEC:V -SPEC:v SPEC:TO/inf
T/prt 	:: ARG Fin T EF 		!COMP:* COMP:v COMP:V COMP:T/prt -COMP:NEG !PROBE:V -SPEC:C/fin -SPEC:T/fin -SPEC:FORCE -SPEC:V
T 		:: ARG 					COMP:* COMP:v COMP:V COMP:Pass -COMP:φ -COMP:D -COMP:NEG !PROBE:V -COMP:Neg/fin -SPEC:T/fin -SPEC:FORCE -SPEC:V -SPEC:v -SPEC:NEG
Neg/fin :: ARG EF Fin NEG  +ΦLF,ΦPF	!ΦPF COMP:T/prt COMP:T -COMP:T/fin -SPEC:T -SPEC:T/fin %SPEC:Inf SEM:internal
v   	:: θ ARG -ΦPF ASP 		!COMP:* COMP:V -SPEC:v -SPEC:T/fin !COMP:V SPEC:φ -SPEC:N
D   	:: φ -ARG 				!COMP:* -COMP:D -COMP:v -COMP:V -COMP:P -COMP:T/fin -COMP:Neg/fin -COMP:T -SPEC:MA.ade/inf -SPEC:Neg -SPEC:P -SPEC:C -SPEC:Inf -SPEC:V COMP:n COMP:N -COMP:T/fin -SPEC:T/fin -SPEC:Neg/fin -COMP:uR !PROBE:N
D/rel	:: φ D -ARG OP OP:REL 	 adjoinable	COMP:* -COMP:C/fin -COMP:φ -SPEC:MA.ade/inf -SPEC:P -SPEC:C -SPEC:Inf -SPEC:V COMP:N -COMP:T/fin -SPEC:T/fin -SPEC:Neg/fin -COMP:uR !PROBE:N
N   	:: θ 					-COMP:EXPL -COMP:VA/inf -COMP:T/fin -COMP:MA/A -COMP:Neg -COMP:T/prt COMP:D/rel COMP:√ COMP:P -COMP:A COMP:C -COMP:AUX -SPEC:FORCE SPEC:A COMP:R COMP:R/D SPEC:φ -SPEC:uWH -SPEC:R -SPEC:V -SPEC:P -SPEC:N -SPEC:C/fin -SPEC:T/fin COMP:uR -SPEC:Adv -COMP:WH
V   	:: θ ARG -ΦPF ASP 		-SPEC:T/fin -SPEC:FORCE SPEC:Adv -COMP:T/prt -COMP:N -COMP:v SPEC:P SPEC:TO/inf -COMP:EXPL
P   	:: ARG 					adjoinable COMP:φ -COMP:V -COMP:MA/A -SPEC:Neg -COMP:Neg -COMP:Adv -SPEC:Adv -COMP:T/fin -COMP:T -COMP:A -SPEC:iR -SPEC:iWH -SPEC:C -SPEC:C/fin -SPEC:Adv -SPEC:T/fin -SPEC:Neg/fin -SPEC:N -SPEC:P -SPEC:T/prt -SPEC:V -SPEC:FORCE -SPEC:v -COMP:uWH
Adv 	:: adjoinable			-SPEC:N -SPEC:FORCE -SPEC:Neg/fin -SPEC:T/fin
A 		:: θ		 			-SPEC:A
Inf		:: TAM 					-COMP:A -COMP:T/fin -COMP:FORCE -COMP:C/fin -SPEC:T/fin -SPEC:V
n		:: φ EF ARG Inf 		N -COMP:n !COMP:* -COMP:IMPSS
√		:: SPEC:φ 2SPEC 		-COMP:φ
Num		:: φ 					COMP:φ COMP:N COMP:n -COMP:D -COMP:QN
DET		:: φ 					COMP:φ -COMP:DET 
Qn		:: φ 					COMP:φ COMP:N COMP:Num
φ		:: θ 					-SPEC:T -SPEC:P COMP:D/rel !COMP:* COMP:N
a		:: 						A !COMP:* -COMP:IMPSS -COMP:a -COMP:A/inf -COMP:T
Neg		:: 						-SPEC:N SPEC:D
ARG		:: PHI:NUM:_ PHI:PER:_ PHI:DET:_

# Universal operator rules
OP				:: -COMP:MA/A
OP:WH			::	DIS:WH OP
WH  			:: -COMP:C/fin -COMP:Adv
OP:REL   		:: adjoinable -COMP:C/fin SPEC:D
FORCE:OP:REL	:: adjoinable
Q				:: OP:WH OP:Q
OP Fin			:: 1EDGE
OP:WH Fin		:: FORCE:WH
OP:Q Fin		:: FORCE:WH
OP:WH			:: DIS:WH OP
OP:Q			:: DIS:WH OP

# Verb classess
V/TR 		:: -COMP:P COMP:φ -COMP:Adv
V/INTR 		:: -COMP:φ -COMP:EXPL SPEC:φ COMP:P
CLASS/DITR 		:: !COMP:* SPEC:φ COMP:φ
PHI_N			:: -COMP:*
ε				:: -ΦPF θ SPEC:φ !wCOMP:θ -wCOMP:P

# Nouns and nounlike elements
R:REF:PROPER_NAME :: -COMP:Inf

# Expletives
T Fin			:: 	SPEC:EXPL

# Generic case features
[NOM]				:: TAIL:T/fin
[ACC]				:: TAIL:v
[PAR]				:: TAIL:v
[GEN]				:: TAIL:Inf
