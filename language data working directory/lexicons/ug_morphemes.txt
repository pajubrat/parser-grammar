
# ======================================================================================
# Universal language-invariant morphemes are collected here
# =======================================================================================

T/fin   :: T/fin PF:T LF:T
v       :: v PF:v LF:v
C/fin   :: C/fin PF:C LF:C/fin 
T		:: T PF:T LF:T FIN
T/prt 	:: T/prt PF:T/prt LF:T/prt
a0		:: A PF:A LF:A
n		:: N n PF:n LF:n
impass	:: v ARG PF:impass LF:impass 
cau		:: v PF:Cau LF:v/Cause
φ		:: φ PF:φ LF:φ VAL ARG INF

# Infinitivals
INF			:: INF PF:INF LF:INF
prtcpl		:: INF T T/prt PF:T/prt LF:T/prt

# English
TO/inf 		:: INF TO/inf PF:TO/inf LF:TO/inf -SPEC:T -COMP:T/fin -SPEC:v -VAL ?ARG adjoinable

# Finnish infinitivals
A/inf		:: A/inf 	INF -PHI 	?ARG	 								PF:(t)A LF:A/inf -SPEC:N -SPEC:Neg/fin SEM/desired_event LANG:FI
VA/inf  	:: VA/inf 	INF T    VAL ARG  	!SPEC:* 						PF:vAn LF:VA -SPEC:N LANG:FI
ESSA/inf 	:: ESSA/inf	INF ADV  VAL ARG 			TAIL:T,$NO_W 			PF:essa LF:while LANG:FI
TUA/inf		:: TUA/inf	INF ADV  VAL ARG 	!SPEC:* TAIL:T,$NO_W 			PF:tua LF:after LANG:FI
KSE/inf		:: KSE/inf	INF ADV  VAL ARG 			TAIL:T,$NO_W 			PF:kse LF:in_order_to LANG:FI
E/inf		:: E/inf 	INF ADV -PHI ARG 	-SPEC:*	TAIL:T,$NO_W 			PF:en LF:by LANG:FI
MA/ssa		:: MA/ssa	INF ADV -PHI ARG 	-SPEC:* TAIL:CLASS/TR,$NO_W 	PF:massa LF:in_doing LANG:FI
MA/tta		:: MA/tta	INF ADV -PHI ARG 	-SPEC:* TAIL:T,$NO_W 			PF:matta LF:without_doing LANG:FI
MA/lla		:: MA/lla	INF ADV -PHI ARG 	-SPEC:* TAIL:T,$NO_W 			PF:malla LF:by_doing COMP:v COMP:V LANG:FI
MA/sta		:: MA/sta	INF ADV -PHI ARG 	-SPEC:* TAIL:CLASS/TR,SEMCLASS_denied_event,$NO_W PF:masta LF:without_doing LANG:FI
MA/an		:: MA/an	INF		-PHI ARG 	-SPEC:* TAIL:CLASS/TR,SEMCLASS_desired_event,$NO_W PF:maan LF:to_do SEM/desired_event LANG:FI

# Italian
ARE/inf		:: INF ARE/inf PF:are LF:ARE/inf ?ARG SPEC:* -SPEC:T/fin VAL -adjoinable 

# Copula
BE-		:: V COPULA PF:BE LF:BE -ARG COMP:φ SPEC:φ !COMP:* COMP:T/prt -COMP:T/fin

# D-elements
D		:: φ PF:D LF:D R:NEW:$$$/SUR
D/rel	:: D/rel φ PF:D/rel LF:D/rel adjoinable
D/pro	:: D φ PF:D LF:D adjoinable R:NEW:REF -R:NEW:$$$/SUR
D/refl 	:: D φ PF:D LF:D adjoinable R:OLD:REF -R:NEW:$$$/SUR
P		:: P PF:P LF:P

# Pronominal items
pro		:: pro-#D
pro-	:: PF:(pro) LF:pro N PHI
cl		:: inflectional CL LEFT:T/fin RIGHT:ARE/inf RIGHT:PRG LEFT:D,CL LANG:IT
inc		:: inflectional INCORPORATED

# Phi- and R-features
sg 			:: inflectional PHI:NUM:SG
pl 			:: inflectional PHI:NUM:PL
1p 			:: inflectional PHI:PER:1
2p 			:: inflectional PHI:PER:2
3p 			:: inflectional PHI:PER:3
def 		:: inflectional PHI:DET:DEF
indef 		:: inflectional PHI:DET:INDEF
hum 		:: inflectional PHI:HUM:HUM
nonhum 		:: inflectional PHI:HUM:NONHUM
m			:: inflectional PHI:GEN:M
f			:: inflectional PHI:GEN:F
proper_name	:: inflectional R:REF:NAME R:NEW:$$$/SUR
r_expression	:: inflectional R:REF:NAME R:NEW:$$$/SUR
pronoun1	:: inflectional R:NEW:REF
reflexive	:: inflectional R:OLD:REF
Rexpression	:: inflectional R:NEW:$$$/SUR
demanaphora	:: inflectional R:OLD:$$$/SUR

# Generic deficient structural cases
nom   	:: inflectional NOM TAIL:FIN,!COMP:*,VAL,$NO_S -FLOAT -adjoinable
acc		:: inflectional ACC TAIL:v,$NO_S
dat		:: inflectional DAT TAIL:T LANG:IT
acc_it	:: inflectional TAIL:v,$NO_S LANG:IT

#Aspect markers
asp		:: inflectional ASP:BOUNDED

# Universal operator feature inventory

pol		:: inflectional OP:POL DIS:POL
C/op	:: inflectional OP:C/OP DIS:C/OP
d/rel   :: inflectional OP:REL
