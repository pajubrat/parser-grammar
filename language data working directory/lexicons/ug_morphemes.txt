
# ======================================================================================
# Universal language-invariant morphemes are collected here
# =======================================================================================

T/fin   :: T/fin PF:T LF:T
v       :: v PF:v LF:v
C/fin   :: C/fin PF:C LF:C/fin 
T		:: T PF:T LF:T FIN
T/prt 	:: T/prt PF:T/prt LF:T/prt
a0		:: A PF:A LF:A
n		:: N n PF:n LF:n
impass	:: v ARG PF:impass LF:impass 
cau		:: v PF:Cau LF:v/Cause
φ		:: φ PF:φ LF:φ VAL ARG INF

# Infinitivals
INF			:: INF PF:INF LF:INF
prtcpl		:: INF T T/prt PF:T/prt LF:T/prt
# English
TO/inf 		:: INF TO/inf PF:TO/inf LF:TO/inf -COMP:T/fin -SPEC:v -VAL ?ARG adjoinable
# Finnish infinitivals
A/inf		:: INF A/inf PF:A/inf LF:A/inf SPEC:* -SPEC:N -SPEC:Neg/fin ?ARG -VAL
ARE/inf		:: INF ARE/inf PF:are LF:ARE/inf ?ARG SPEC:* -SPEC:T/fin VAL -adjoinable 
ESSA/inf 	:: ADV INF ESSA/inf ?VAL ?ARG SPEC:* TAIL:T PF:essa LF:while
TUA/inf		:: ADV INF TUA/inf VAL TAIL:T ARG !SPEC:* PF:tua LF:after
VA/inf  	:: VAL ARG INF VA/inf PF:vAn LF:VA -SPEC:N SPEC:*
MA/ssa		:: MA/inf MA/ssa ADV COMP:v -SPEC:* COMP:V PF:massa LF:in_doing TAIL:V -VAL ARG adjoinable 
MA/tta		:: MA/inf ADV PF:matta LF:without_doing -VAL ARG TAIL:T adjoinable
MA/lla		:: MA/inf ADV PF:malla LF:by_doing TAIL:T COMP:v COMP:V VAL ARG adjoinable
MA/inf		:: MA/inf ADV PF:maan LF:to_do TAIL:V COMP:v COMP:V VAL ARG
E/inf		:: PF:en LF:by ADV -ARG -VAL TAIL:T LANG:FI

# Copula
BE-		:: V COPULA PF:BE LF:BE -ARG COMP:φ SPEC:φ !COMP:* COMP:T/prt -COMP:T/fin
D		:: D φ PF:D LF:D
D/rel	:: D/rel φ PF:D/rel LF:D/rel adjoinable
P		:: P PF:P LF:P

# Pronominal items
pro		:: pro-#D
pro-	:: PF:(pro) LF:pro N PHI
cl		:: inflectional CL LEFT:T/fin RIGHT:ARE/inf RIGHT:PRG LEFT:D,CL
inc		:: inflectional INCORPORATED

# Phi- and R-features
sg 			:: inflectional PHI:NUM:SG
pl 			:: inflectional PHI:NUM:PL
1p 			:: inflectional PHI:PER:1
2p 			:: inflectional PHI:PER:2
3p 			:: inflectional PHI:PER:3
def 		:: inflectional PHI:DET:DEF
indef 		:: inflectional PHI:DET:INDEF
hum 		:: inflectional PHI:HUM:HUM
nonhum 		:: inflectional PHI:HUM:NONHUM
m			:: inflectional PHI:GEN:M
f			:: inflectional PHI:GEN:F
proper_name	:: inflectional R:REF:NAME R:NEW:$$$/SUR
pronoun1	:: inflectional R:NEW:REF
reflexive	:: inflectional R:OLD:REF
Rexpression	:: inflectional R:NEW:$$$/SUR
demanaphora	:: inflectional R:OLD:$$$/SUR

# Generic deficient structural cases
nom   	:: inflectional NOM TAIL:FIN,!COMP:*,VAL -FLOAT -adjoinable
acc		:: inflectional ACC TAIL:v
dat		:: inflectional DAT TAIL:v
acc_it	:: inflectional TAIL:v

#Aspect markers
asp		:: inflectional ASP:BOUNDED

# Universal operator feature inventory
uC/op	:: inflectional uC/op
wh     	:: inflectional OP:WH
WH		:: inflectional FORCE:WH
wh_		:: inflectional OP
top    	:: inflectional OP:TOP DIS:TOP
foc    	:: inflectional OP:FOC DIS:FOC
fam		:: inflectional OP:FAM DIS:FAM
pol		:: inflectional OP:POL DIS:POL
C/op	:: inflectional OP:C/OP DIS:C/OP
d/rel   :: inflectional OP:REL
Q		:: inflectional OP OP:WH OP:Q FORCE:WH DIS:WH

# Miscellanneus items
# Possessive, as in John's, but status unclear
poss	:: inflectional POSS
