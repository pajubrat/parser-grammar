
# ======================================================================================

T/fin   :: T/fin PF:T LF:tense
v       :: v PF:-ø- LF:v
C/fin   :: C/fin PF:C LF:C/fin
T		:: T PF:T LF:tense Fin EF:φ
T/prt 	:: T/prt PF:T/prt LF:prtcpl
a0		:: A PF:A LF:mod
n		:: N n PF:n LF:thing
impass	:: v ARG PF:impass LF:impass 
cau		:: v PF:Cau LF:cause
φ		:: φ PF:φ LF:phi ARG Inf
Agr		:: Agr !EF:φ PF:Agr LF:Agr ARG Fin

# Infinitivals
Inf			:: PF:Inf LF:inf
prtcpl		:: Inf T T/prt PF:T/prt LF:prtcpl

# English

# Finnish infinitivals

ESSA/inf 	:: ESSA/inf		Inf Adv ARG   EF:φ TAIL:T,$NO_W 	PF:-essA 	LF:while LANG:FI
TUA/inf		:: TUA/inf		Inf Adv ARG  !EF:φ TAIL:T,$NO_W 	PF:-tUA 	LF:after LANG:FI
KSE/inf		:: KSE/inf		Inf Adv ARG   EF:φ TAIL:T,$NO_W 	PF:-kse 	LF:for LANG:FI
E/inf		:: E/inf 		Inf Adv ARG  -EF:φ TAIL:T,$NO_W 	PF:-en 		LF:by LANG:FI
MA.ine/inf	:: MA.ine/inf	Inf Adv ARG  -EF:φ TAIL:V,$NO_W 	PF:-mAssA 	LF:in LANG:FI
MA.abe/inf	:: MA.abe/inf	Inf Adv ARG  -EF:φ TAIL:T,$NO_W 	PF:-mAttA 	LF:without LANG:FI

# Prticiple adjectives

# Italian

# Copula

# D-elements
D		:: φ D PF:D LF:det R:NEW:$$$/SUR

# Pronominal items
