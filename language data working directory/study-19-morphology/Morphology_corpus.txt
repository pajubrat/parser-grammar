
& 2. Syntactic opaqueness

	& 2.1 Verbal inner layer: causatives, reflexives, frequentatives, semelfactive, thematic vowel

		& 2.1.1 Causatives
	
			# Simple bimorphemic causative
			
				Sirkku kuiva-#TA#v#T/fin#prs#[V] sukkaa
				'Sirkku.nom dry.cau.prs.3sg sock.par'

			# Causative without direct object is ungrammatical

				*Sirkku kuiva-#TA#v#T/fin#prs#[V]
				'Sirkku.nom dry.cau.prs.3sg'

			# Absent v, the sentence is ungrammatical (no case checking plus projection principle violation)

				*Sirkku kuiva-#TA#T/fin#prs#[V] sukkaa
				'Sirkku.nom dry.cau.prs.3sg sock.par'

			# Stranded T is ungrammatical

				*Sirkku T/fin#prs#[V] kuiva-#TA#v sukkaa
				'Sirkku.nom prs.3sg dry.cau sock.par'

			# Double causative

				Sirkku kuiva-#TA#TA#v#T/fin#prs#[V] sukkaa
				'Sirkku.nom dru.cau.cau.prs.3sg sock.par'

			# Denominal and deadjectival causatives
			
				kenkä-#TA#minen	
				'shoe.cau.abs'  /kengittäminen/

				lyhyt-#TA#minen
				'short.cau.abs'	/lyhentäminen/

				Sirkku pino-#TA#v#T/fin#prs#[V] sukkaa
				'Sirkku stack(N).cau.v.prs.3sg sock.par'

				Sirkku pino-#TA#TA#v#T/fin#prs#[V] sukkaa
				'Sirkku stack(N).cau.cau.v.prs.3sg sock.par'

				Sirkku pino-#TA#TA#TA#v#T/fin#prs#[V] sukkaa
				'Sirkku stack(N).cau.cau.cau.v.prs.3sg sock.par'

		& 2.1.2 Reflexives
		
			# Simple bimorphemic reflexive
			
				sukka kuiva-#UTU#T/fin#prs#[V] 
				'sock.nom dry-refl.prs.3sg'
				
			# Ungrammatical with direct object
			
				*Sirkku kuiva-#UTU#T/fin#prs#[V] sukkaa
				'Sirkku.nom dry.refl.prs.3sg sock.par'
				
			# Ungrammatical with small verb v

	& 2.2 Verbal outer layer
	
		& 2.2.1 Impersonal passive
		
			# Simple impersonal passive
			
				peli pelattiin
				'game.acc0 play.pst.impss'
				
				peli pelaa-#impss#T/fin#Vn
				'game.acc0 play-pst-impss'

				Helsingissä pelattiin peli
				'Helsinki-ine play.pst.impss game.acc0'
		
			# Ungrammatical with external argument except 1pl
			
				*sinä pelattiin peli
				'you.nom play.pst.impss game.acc0
		
				me pelattiin peli
				'we.nom play.pst.impss game.acc0'
			
			# With inner layer morphemes
			
				peli pelaa-#TA#impss#T/fin#Vn
				'game.acc0 play-cau-impss-prs.pl'
			
				peli pelaa-#TA#TA#UTU#impss#T/fin#Vn
				'game.acc0 play-cau-cau-refl-impss-prs.pl'
				
			# Ungrammatical if inside inner layer
			
				*peli pelaa-#impss#TA#TA#UTU#T/fin#Vn
				'game.acc0 play-cau-cau-refl-impss-prs.pl'

			# Syntactic selection test violations
			
				*peli pelaa-#impss#impss#T/fin
				'game play-impss-impss-prs.pl'

				*peli pelaa-#T/fin#impss
				'game play-prs-impss-pl

& 3. Stranding 

	# Verbal causatives
	
		# Causative stacking should not stack external/internal postverbal arguments
		
			Sirkku kuiva-#TA#v#T/fin#prs#[V] sukkaa
			'Sirkku.nom dry.cau.prs.3sg sock.par'

			*Sirkku kuiva-#TA#TA#v#T/fin#prs#[V] Pekka sukkaa
			'Sirkku.nom dry.cau.prs.3sg Pekka.nom sock.par'
			#Intended: 'Sirkku causes Pekka to dry the sock.'

			*Sirkku kuiva-#TA#TA#v#T/fin#prs#[V] Pekan_acc sukkaa
			'Sirkku.nom dry.cau.prs.3sg Pekka.acc sock.par'
			#Intended: 'Sirkku causes Pekka to dry the sock.'
			
			*Sirkku kuiva-#TA#TA#TA#v#T/fin#prs#[V] Pekka Pekka sukkaa
			'Sirkku.nom dry.cau.prs.3sg Pekka.nom Pekka.nom sock.par'
			#Intended: 'Sirkku causes Pekka1 to cause Pekka2 to dry the sock.'
		
		# Causatives wrapped insided reflexives should not introduce external/internal postverbal arguments
		
			sukka kuiva-#TA#TA#UTU#T/fin#prs#[V]
			'sock.nom dry-cau-cau-refl-prs.3sg'
			
			*Sirkku kuiva-#TA#TA#UTU#T/fin#prs#[V] sukkaa
			'Sirkku.nom dry-cau-cau-refl-prs.3sg'

			*Sirkku kuiva-#TA#TA#UTU#T/fin#prs#[V] sukan_acc
			'Sirkku.nom dry-cau-cau-refl-prs.3sg'

	# Denominal causatives and stranding

		# Verify that nominal/adjectival modifiers cannot modify the causativised stem to rule out IHM/Cau-NP analysis
	
			Sirkku kenkä-#TA#v#T/fin#prs#[V] Pekan_gen hevosen
			'Sirkku shoe-cau-v-prs-3sg Pekka.gen horse.acc'
			'*Sirkku causes the horse to have Pekka's shoes.'
				
			Sirkku kenkä-#TA#v#T/fin#prs#[V] pienen hevosen
			'Sirkku.nom shoe-cau-v-prs-3sg small.acc horse.acc'
			'*Sirkku causes the horse to have small shoes.'
			
			Sirkku kenkä-#TA#v#T/fin#prs#[V] tämän hevosen/2
			'Sirkku.nom shoe-cau-v-prs-3sg this.acc horse.acc'
			'*Sirkku causes the horse to have this shoe.'
		
		# Verify that postverbal nominal modifiers are ungrammatical in denominal intransitives, ruling out IHM/Refl-NP analysis

			Sirkku kenkä-#TA#UTU#T/fin#prs#[V]
			'Sirkku.nom shoe-cau-relf-prs-3sg'

			*Sirkku kenkä-#TA#UTU#T/fin#prs#[V] Pekan_gen
			'Sirkku.nom shoe-cau-relf-prs-3sg Pekka.gen'
			#Intended: 'Sirkku is caused to have Pekka's shoes'
			
			*Sirkku kenkä-#TA#UTU#T/fin#prs#[V] pienen
			'Sirkku.nom shoe-cau-relf-prs-3sg small.acc'
			#Intended: 'Sirkku is caused to have small shoes'

			*Sirkku kenkä-#TA#UTU#T/fin#prs#[V] tämän
			'Sirkku.nom shoe-cau-relf-prs-3sg this.acc (this=D^0, not DP)'
			#Intended: 'Sirkku is caused to have this shoe.'
			
& 4. Free/bound morpheme tests

	& 4.1 Verbal morphemes

		& T/fin + inflection
		
			*Sirkku T/fin#prs#[V] laula-
			'Sirkku prs.3sg sing'
			
			Sirkku laula-#T/fin#prs#[V]
			'Sirkku.nom sing-prs.3sg'
		
		& Third person singular inflection
		
			*Sirkku [V] laula-#T/fin#prs
			'Sirkku.nom 3sg sing-prs'
		
			*Sirkku [V] T/fin#prs laula-
			'Sirkku.nom 3sg prs sing'
			
		& Causative
		
			*Sirkku T/fin#prs#[V] TA laula-
			'Sirkku.nom prs.3sg cau sing'
			
			Sirkku laula-#TA#T/fin#prs#[V] Merjaa
			'Sirkku.nom sing-cau-prs-3sg Merja.par'
		
		& Frequentive
		
			*Sirkku T/fin#prs#[V] skele laula-
			'Sirkku prs.3sg freq sing
			
			Sirkku laula-#skele#T/fin#prs#[V]
			'Sirkku.nom sing-freq-prs-3sg'
		
		& Reflexive
				
			*sukka T/fin#prs#[V] U kuiva-
			'sock.nom prs.3sg relf dry'
			
			sukka kuiva-#U#T/fin#prs#[V]
			'sock dry-refl-prs-3sg'

		& small verb v

			*Sirkku T/fin#prs#[V] v kuiva- sukkaa
			'Sirkku.nom prs.3sg v dry sock.par'
			
			Sirkku kuiva-#v#T/fin#prs#[V] sukkaa
			'Sirkku.nom dry-prs-3sg sock.par'
		
		& A-infinitive
		
			*Sirkku haluaa (t)A laula-
			'Sirkku.nom want.prs.3sg A/inf sing'
			
			Sirkku haluaa laula-#(t)A
			'Sirkku.nom want.prs.3sg sing-a/inf'
